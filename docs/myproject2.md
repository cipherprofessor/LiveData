def find_matches(user):
    # 1. Extract user's skills to teach and learn
    user_teaches = user.skillsToTeach
    user_learns = user.skillsToLearn
    
    # 2. Find users with complementary skills
    potential_matches = db.query(
        "SELECT * FROM users WHERE 
         skillsToTeach IN user_learns AND
         skillsToLearn IN user_teaches AND
         location_distance < 10km"
    )
    
    # 3. Calculate match score
    for match in potential_matches:
        score = 0
        
        # Skill overlap (30%)
        skill_overlap = count_matching_skills(user, match)
        score += skill_overlap * 0.3
        
        # Location proximity (20%)
        distance = calculate_distance(user.location, match.location)
        score += (10 - distance) * 0.2
        
        # Rating/reputation (25%)
        score += match.rating * 0.25
        
        # Activity level (15%)
        score += match.swapsCompleted * 0.15
        
        # Availability match (10%)
        availability_overlap = check_calendar_overlap(user, match)
        score += availability_overlap * 0.1
        
        match.score = score
    
    # 4. Return top 10 matches sorted by score
    return sorted(potential_matches, key=lambda x: x.score, reverse=True)[:10]

typescriptimport { PrismaClient, User, UserSkill, Skill } from '@prisma/client';
import { calculateDistance } from '../lib/utils';

const prisma = new PrismaClient();

interface MatchScore {
  userId: string;
  user: User;
  score: number;
  breakdown: {
    skillMatch: number;
    locationProximity: number;
    rating: number;
    activityLevel: number;
    availability: number;
  };
  matchedSkills: {
    theyTeach: Skill[];
    youNeed: Skill[];
    youTeach: Skill[];
    theyNeed: Skill[];
  };
}

export class SkillMatcher {
  /**
   * Find best matches for a user based on complementary skills
   */
  async findMatches(userId: string, limit: number = 10): Promise<MatchScore[]> {
    // 1. Get current user's profile
    const currentUser = await prisma.user.findUnique({
      where: { userId },
      include: {
        skillsToTeach: {
          where: { canTeach: true },
          include: { skill: true },
        },
        skillsToLearn: {
          where: { wantsToLearn: true },
          include: { skill: true },
        },
      },
    });

    if (!currentUser) throw new Error('User not found');

    const teachSkillIds = currentUser.skillsToTeach.map((us) => us.skillId);
    const learnSkillIds = currentUser.skillsToLearn.map((us) => us.skillId);

    // 2. Find potential matches
    // Users who teach what I want to learn AND want to learn what I teach
    const potentialMatches = await prisma.user.findMany({
      where: {
        AND: [
          { userId: { not: userId } },
          { isActive: true },
          {
            skillsToTeach: {
              some: {
                canTeach: true,
                skillId: { in: learnSkillIds },
              },
            },
          },
          {
            skillsToLearn: {
              some: {
                wantsToLearn: true,
                skillId: { in: teachSkillIds },
              },
            },
          },
        ],
      },
      include: {
        skillsToTeach: {
          where: { canTeach: true },
          include: { skill: true },
        },
        skillsToLearn: {
          where: { wantsToLearn: true },
          include: { skill: true },
        },
      },
      take: 100, // Get more, then filter
    });

    // 3. Calculate match scores
    const scoredMatches: MatchScore[] = [];

    for (const match of potentialMatches) {
      const score = this.calculateMatchScore(currentUser, match);
      
      if (score.score > 30) { // Minimum threshold
        scoredMatches.push(score);
      }
    }

    // 4. Sort by score and return top matches
    return scoredMatches
      .sort((a, b) => b.score - a.score)
      .slice(0, limit);
  }

  /**
   * Calculate comprehensive match score between two users
   */
  private calculateMatchScore(
    user1: any,
    user2: any
  ): MatchScore {
    const breakdown = {
      skillMatch: 0,
      locationProximity: 0,
      rating: 0,
      activityLevel: 0,
      availability: 0,
    };

    // 1. SKILL MATCH (40% weight)
    const user1TeachSkills = user1.skillsToTeach.map((us: any) => us.skillId);
    const user1LearnSkills = user1.skillsToLearn.map((us: any) => us.skillId);
    const user2TeachSkills = user2.skillsToTeach.map((us: any) => us.skillId);
    const user2LearnSkills = user2.skillsToLearn.map((us: any) => us.skillId);

    // Skills user2 teaches that user1 wants to learn
    const skillsTheyTeach = user2.skillsToTeach.filter((us: any) =>
      user1LearnSkills.includes(us.skillId)
    );

    // Skills user1 teaches that user2 wants to learn
    const skillsTheyNeed = user2.skillsToLearn.filter((us: any) =>
      user1TeachSkills.includes(us.skillId)
    );

    const complementarySkills = skillsTheyTeach.length + skillsTheyNeed.length;
    breakdown.skillMatch = Math.min(complementarySkills * 10, 40); // Max 40 points

    // 2. LOCATION PROXIMITY (20% weight)
    if (user1.latitude && user1.longitude && user2.latitude && user2.longitude) {
      const distance = calculateDistance(
        user1.latitude,
        user1.longitude,
        user2.latitude,
        user2.longitude
      );
      
      // Closer = better (within 10km is best)
      if (distance < 5) breakdown.locationProximity = 20;
      else if (distance < 10) breakdown.locationProximity = 15;
      else if (distance < 20) breakdown.locationProximity = 10;
      else if (distance < 50) breakdown.locationProximity = 5;
    } else if (user1.city === user2.city) {
      breakdown.locationProximity = 15; // Same city
    } else if (user1.state === user2.state) {
      breakdown.locationProximity = 5; // Same state
    }

    // 3. RATING/REPUTATION (20% weight)
    const averageRating = (user2.rating / 5) * 20;
    breakdown.rating = Math.round(averageRating);

    // 4. ACTIVITY LEVEL (15% weight)
    const swapSuccessRate = user2.totalSwaps > 0
      ? (user2.completedSwaps / user2.totalSwaps) * 15
      : 0;
    breakdown.activityLevel = Math.round(swapSuccessRate);

    // 5. AVAILABILITY (5% weight)
    // Check last active (active in last 7 days = 5 points)
    const daysSinceActive = Math.floor(
      (Date.now() - user2.lastActive.getTime()) / (1000 * 60 * 60 * 24)
    );
    if (daysSinceActive <= 7) breakdown.availability = 5;
    else if (daysSinceActive <= 30) breakdown.availability = 3;
    else breakdown.availability = 1;

    // Calculate total score
    const totalScore =
      breakdown.skillMatch +
      breakdown.locationProximity +
      breakdown.rating +
      breakdown.activityLevel +
      breakdown.availability;

    return {
      userId: user2.userId,
      user: user2,
      score: totalScore,
      breakdown,
      matchedSkills: {
        theyTeach: skillsTheyTeach.map((us: any) => us.skill),
        youNeed: user1.skillsToLearn
          .filter((us: any) => skillsTheyTeach.some((st: any) => st.skillId === us.skillId))
          .map((us: any) => us.skill),
        youTeach: user1.skillsToTeach
          .filter((us: any) => skillsTheyNeed.some((sn: any) => sn.skillId === us.skillId))
          .map((us: any) => us.skill),
        theyNeed: skillsTheyNeed.map((us: any) => us.skill),
      },
    };
  }

  /**
   * Get skill recommendations based on user's current skills
   */
  async getSkillRecommendations(userId: string, limit: number = 10): Promise<Skill[]> {
    const user = await prisma.user.findUnique({
      where: { userId },
      include: {
        skillsToTeach: { include: { skill: { include: { category: true } } } },
        skillsToLearn: { include: { skill: true } },
      },
    });

    if (!user) throw new Error('User not found');

    const existingSkillIds = [
      ...user.skillsToTeach.map((us) => us.skillId),
      ...user.skillsToLearn.map((us) => us.skillId),
    ];

    // Get popular skills in same categories
    const userCategories = user.skillsToTeach.map((us) => us.skill.category.categoryId);

    const recommendations = await prisma.skill.findMany({
      where: {
        AND: [
          { skillId: { notIn: existingSkillIds } },
          { categoryId: { in: userCategories } },
          { isActive: true },
        ],
      },
      orderBy: { popularity: 'desc' },
      take: limit,
      include: { category: true },
    });

    return recommendations;
  }
}

typescript/**
 * Calculate distance between two coordinates using Haversine formula
 * Returns distance in kilometers
 */
export function calculateDistance(
  lat1: number,
  lon1: number,
  lat2: number,
  lon2: number
): number {
  const R = 6371; // Earth's radius in km
  const dLat = toRad(lat2 - lat1);
  const dLon = toRad(lon2 - lon1);

  const a =
    Math.sin(dLat / 2) * Math.sin(dLat / 2) +
    Math.cos(toRad(lat1)) *
      Math.cos(toRad(lat2)) *
      Math.sin(dLon / 2) *
      Math.sin(dLon / 2);

  const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
  const distance = R * c;

  return Math.round(distance * 10) / 10; // Round to 1 decimal
}

function toRad(degrees: number): number {
  return degrees * (Math.PI / 180);
}

  // Level 1: 0-99 XP

  // Formula: Level = floor(sqrt(XP/100)) + 1
  return Math.floor(Math.sqrt(experiencePoints / 100)) + 1;

 */

  if (minutes < 60) return `${minutes}m`;
  
  const hours = Math.floor(minutes / 60);
  const mins = minutes % 60;
  
  if (mins === 0) return `${hours}h`;
  return `${hours}h ${mins}m`;
}

/**
 * Helper: Check and award badges
 */
async function checkAndAwardBadges(userId: string) {
  const user = await prisma.user.findUnique({
    where: { userId },
    include: {
      badges: { include: { badge: true } },
    },
  });

  if (!user) return;

  const allBadges = await prisma.badge.findMany({ where: { isActive: true } });
  const earnedBadgeIds = user.badges.map((ub) => ub.badgeId);

  for (const badge of allBadges) {
    // Skip if already earned
    if (earnedBadgeIds.includes(badge.badgeId)) continue;

    // Check requirements
    const requirements = JSON.parse(badge.requirement);
    let meetsRequirements = true;

    for (const [key, value] of Object.entries(requirements)) {
      if (user[key as keyof typeof user] === undefined || user[key as keyof typeof user] < value) {
        meetsRequirements = false;
        break;
      }
    }

    if (meetsRequirements) {
      // Award badge
      await prisma.userBadge.create({
        data: {
          userId,
          badgeId: badge.badgeId,
          progress: 100,
        },
      });

      // Award SkillCoins
      await prisma.user.update({
        where: { userId },
        data: { skillCoins: { increment: badge.skillCoins } },
      });

      // Notify user
      await prisma.notification.create({
        data: {
          userId,
          type: 'BADGE_EARNED',
          title: 'New Badge Earned!',
          message: `You earned the "${badge.name}" badge! +${badge.skillCoins} SkillCoins`,
          actionUrl: '/profile/badges',
        },
      });
    }
  }
}

/**
 * GET /api/v1/chat/:userId
 * Get conversation with a user
 */
router.get('/:userId', authenticate, async (req: AuthenticatedRequest, res, next) => {
  try {
    const { userId: currentUserId } = req.user!;
    const { userId: otherUserId } = req.params;
    const { limit = 50, before } = req.query;

    const where: any = {
      OR: [
        { senderId: currentUserId, receiverId: otherUserId },
        { senderId: otherUserId, receiverId: currentUserId },
      ],
    };

    if (before) {
      where.createdAt = { lt: new Date(before as string) };
    }

    const messages = await prisma.message.findMany({
      where,
      orderBy: { createdAt: 'desc' },
      take: parseInt(limit as string),
      include: {
        sender: {
          select: {
            userId: true,
            firstName: true,
            lastName: true,
            profilePicture: true,
          },
        },
      },
    });

    // Mark messages as read
    await prisma.message.updateMany({
      where: {
        senderId: otherUserId,
        receiverId: currentUserId,
        isRead: false,
      },
      data: {
        isRead: true,
        readAt: new Date(),
      },
    });

    res.json({ messages: messages.reverse() });
  } catch (error) {
    next(error);
  }
});

/**
 * GET /api/v1/chat/conversations
 * Get list of conversations
 */
router.get('/conversations', authenticate, async (req: AuthenticatedRequest, res, next) => {
  try {
    const { userId } = req.user!;

    // Get distinct users the current user has chatted with
    const sentTo = await prisma.message.findMany({
      where: { senderId: userId },
      select: { receiverId: true },
      distinct: ['receiverId'],
    });

    const receivedFrom = await prisma.message.findMany({
      where: { receiverId: userId },
      select: { senderId: true },
      distinct: ['senderId'],
    });

    const userIds = [
      ...new Set([
        ...sentTo.map((m) => m.receiverId),
        ...receivedFrom.map((m) => m.senderId),
      ]),
    ];

    // Get conversation details
    const conversations = await Promise.all(
      userIds.map(async (otherUserId) => {
        const lastMessage = await prisma.message.findFirst({
          where: {
            OR: [
              { senderId: userId, receiverId: otherUserId },
              { senderId: otherUserId, receiverId: userId },
            ],
          },
          orderBy: { createdAt: 'desc' },
        });

        const unreadCount = await prisma.message.count({
          where: {
            senderId: otherUserId,
            receiverId: userId,
            isRead: false,
          },
        });

        const user = await prisma.user.findUnique({
          where: { userId: otherUserId },
          select: {
            userId: true,
            firstName: true,
            lastName: true,
            profilePicture: true,
            lastActive: true,
          },
        });

        return {
          user,
          lastMessage,
          unreadCount,
        };
      })
    );

    // Sort by last message time
    conversations.sort((a, b) => {
      const aTime = a.lastMessage?.createdAt.getTime() || 0;
      const bTime = b.lastMessage?.createdAt.getTime() || 0;
      return bTime - aTime;
    });

    res.json({ conversations });
  } catch (error) {
    next(error);
  }
});
